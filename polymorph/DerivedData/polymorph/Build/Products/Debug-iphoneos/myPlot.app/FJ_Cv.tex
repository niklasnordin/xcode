\[
c_v = - R \tau^2 \left[ \left( \frac{\partial^2 \alpha^0}{\partial \tau^2 } \right)_\delta +  \left( \frac{\partial^2 \bar{\alpha}}{\partial \tau^2 } \right)_\delta\right]
\]
where $\bar \alpha$ is the residual dimensionless Helmholtz energy (see function FJ\_Rho for more info) and
\begin{eqnarray*}
\alpha^0(\delta, \tau) &=& \frac{h_0^0\tau}{RT_c} - \frac{s_0}{R} - 1 + \ln \frac{\tau_0 \delta}{\delta_0 \tau} \\
&+& \frac{\tau}{R}\int_{\tau_0}^{\tau}\frac{c_p^0}{\tau^2} d\tau + \frac{1}{R}\int_{\tau_0}^{\tau} \frac{c_p^0}{\tau} d\tau
\end{eqnarray*}\\
